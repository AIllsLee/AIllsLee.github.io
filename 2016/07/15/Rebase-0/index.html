Rebase

Share

#####1. 更新master到最新
cd project_dir

git checkout master

git pull origin master

#####2. 开始开发task-1186
git checkout -b task-1186

#####3. 提交
git commit -am “for #1186 xxxxx”

//或者

git add .

git commit -m“for #1186 xxxxx”

#####4. rebase最新的master

git checkout master

git pull

git checkout task-1186

git rebase master

//
有冲突的时候 先解决冲突

git add .

git rebase –continue

//

git checkout master

git merge task-1186

#####5. 将master推送到远程
git push

#####6. 删除本地分支
git branch -d task-1186
